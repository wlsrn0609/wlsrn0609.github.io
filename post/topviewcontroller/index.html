WindowHelper#

import UIKit

// Utility class to handle UIWindow related functionality
class WindowHelper {
    
    // This property returns the current key window, considering different iOS versions.
    static var keyWindow: UIWindow? {
        #if swift(>=5.1)
        if #available(iOS 13, *) {
            return UIApplication.shared.connectedScenes
                .compactMap { $0 as? UIWindowScene }
                .flatMap { $0.windows }
                .first(where: { $0.isKeyWindow })
        } else {
            return UIApplication.shared.keyWindow
        }
        #else
        return UIApplication.shared.keyWindow
        #endif
    }
    
    // This method returns the top-most view controller from the key window.
    static func topViewController() -> UIViewController? {
        guard var topController = keyWindow?.rootViewController else {
            return nil
        }
        while let presentedViewController = topController.presentedViewController {
            topController = presentedViewController
        }
        return topController
    }
}

How To Use#

extension UIViewController {
    var topViewController: UIViewController? {
        return WindowHelper.topViewController()
    }
}
